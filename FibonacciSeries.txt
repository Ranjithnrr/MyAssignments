package assignment;

public class W1D1A1FibonacciSeries {

	public static void main(String[] args) {
		
		int firstNum = 0, secNum = 1, sum = 0;
		
		System.out.println(firstNum);
		System.out.println(secNum);
		
		for (int i = 0; i < 11; i++) {
			sum = firstNum + secNum;
			firstNum = secNum;
			secNum = sum;
			
			System.out.println(sum);
		}
		

	}

}
